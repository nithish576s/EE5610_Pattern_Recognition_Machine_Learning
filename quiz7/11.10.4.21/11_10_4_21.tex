\section{Exercise 10.4}
\begin{enumerate}
\item Find the equation of line which is equidistant from parallel lines $9x+6y-7=0$ and $3x+2y+6=0$.

\begin{figure}[ht]
	\centering
	\includegraphics[width = \columnwidth]{figs/line_plot.jpg}
	\caption{Triangle ABC}
	\label{fig:1}
\end{figure}


Equation of line 1 is,

\begin{align}
	\myvec{9 & 6}\vec{x} &= 7\\
	\myvec{3 & 2}\vec{x} &= \frac{7}{3}
\end{align}

Equation of line 2 is,

\begin{align}
	\myvec{3 & 2}\vec{x} &= -6
\end{align}

The equation of the equidistant line will be of the form,

\begin{align}
	\myvec{3 & 2}\vec{x} &= c
\end{align}

If,

\begin{align}
	\vec{n}^{\top}\vec{x} = c_1\\
	\vec{n}^{\top}\vec{x} = c_2
\end{align}
 
are two parallel lines then the distance between these two lines is,

\begin{align}
	d = \frac{\abs{c_1-c_2}}{\norm{\vec{n}}}
\end{align}

We need to find c such that,

\begin{align}
	\frac{\abs{c-c_1}}{\norm{\vec{n}}} = \frac{\abs{c-c_2}}{\norm{\vec{n}}}
\end{align}

where, 

\begin{align}
	\vec{n} = \myvec{3 \\ 2}
\end{align}

\begin{align}
	\frac{\abs{c-c_1}}{\norm{\vec{n}}} &= \frac{\abs{c-c_2}}{\norm{\vec{n}}}\\
	\abs{c-\frac{7}{3}} &= \abs{c-\brak{-6}}\\
	\frac{7}{3} - c &= c + 6\\
	2c &= \frac{7}{3}-6 = \frac{-11}{3}\\
	c &= \frac{-11}{6}
\end{align}

Hence, the equation of the line that is equidistant to parallel lines,

\begin{align}
	\myvec{9 & 6}\vec{x} &= 7\\
	\myvec{3 & 2}\vec{x} &= -6
\end{align}

is,

\begin{align}
	\myvec{3 & 2}\vec{x} &= -\frac{11}{6}
\end{align}

\end{enumerate}
