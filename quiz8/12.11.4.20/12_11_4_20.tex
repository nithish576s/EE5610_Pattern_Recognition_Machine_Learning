\section{Exercise 11.4}
\begin{enumerate}
	\item Find the equation of the line passing through the point $\brak{1,2,-4}$ and perpendicular to the two lines:

\begin{align*}
	\frac{x-8}{3}=\frac{y+19}{-16}=\frac{z-10}{7} \text{ and }\\ \frac{x-15}{3}=\frac{y-29}{8}=\frac{z-5}{-5} 
\end{align*}

The direction vector of the line 
\begin{align}
	\frac{x-8}{3}=\frac{y+19}{-16}=\frac{z-10}{7}
\end{align}

is

\begin{align}
	\vec{m_1} = \myvec{3\\-16\\7}
\end{align}

The direction vector of the line 

\begin{align}
	\frac{x-15}{3}=\frac{y-29}{8}=\frac{z-5}{-5}
\end{align}

is

\begin{align}
	\vec{m_2} = \myvec{3\\8\\-5}
\end{align}

Let $\vec{n}$ denote the direction vector of the line perpendicular to the given two lines.

\begin{align}
	\vec{m_1}^{\top}\vec{n} = 0 \implies \myvec{3 & -16 & 7}\vec{n}=0\\
	\vec{m_2}^{\top}\vec{n} = 0 \implies \myvec{3 & 8 & -5}\vec{n}=0\\
\end{align}

Let us find the $\vec{n}$ that satisfies these equations.

\begin{align}
	\myvec{3 & -16 & 7\\3 & 8 & -5}\vec{n} = 0\\
	\xleftrightarrow[]{R_2\leftarrow R_2-R_1}\\
 	\myvec{3 & -16 & 7\\0 & 24 & -12}\vec{n} = 0\\
	\xleftrightarrow[]{R_1\leftarrow R_1+\frac{2}{3}R_2}\\
	\myvec{3 & 0 & -1\\0 & 24 & -12}\vec{n} = 0\\
	\xleftrightarrow[]{R_2\leftarrow R_2/12}\\
	\myvec{3 & 0 & -1\\0 & 2 & -1}\myvec{n_1\\n_2\\n_3} = 0\\
	3n_1-n_3=0 \implies n_1 = \frac{n_3}{3}\\
	2n_2-n_3=0 \implies n_2 = \frac{n_3}{2}\\
	\implies \vec{n} = \myvec{2\\3\\6}
\end{align}

Hence the vector equation of the line passing through $\brak{1,2,-4}$ is,

\begin{align}
	\vec{x} = \myvec{1\\2\\-4} + \lambda \myvec{2\\3\\6}
\end{align}

\end{enumerate}
